//Write a C program to take two numbers as input from the user and find the maximum between the two numbers using if else.
#include <stdio.h>
int main(){
    float num1,num2;
    printf("Input num1: ");
    scanf(" %f",&num1);
    printf("Input num2: ");
    scanf(" %f",&num2);
    if(num1>num2){
        printf("Maximum: %g",num1);
    }else if(num1==num2){
        printf("Same number bro!");
    }
    else{
        printf("Maximum: %g",num2);
    }
    return 0;
}
